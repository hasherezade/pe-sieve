261 {
262  HMODULE hMods[1024] = { 0 };
263  const size_t modules_count = enum_modules(this->processHandle, hMods, sizeof(hMods), LIST_MODULES_ALL);
264  if (modules_count == 0) {
265  return 0;
266  }
267  if (args.imprec_mode != PE_IMPREC_NONE || args.iat != pesieve::PE_IATS_NONE) {
268  pReport.exportsMap = new peconv::ExportsMapper();
269  }
270 
271  size_t counter = 0;
272  for (counter = 0; counter < modules_count; counter++) {
273  if (processHandle == nullptr) break;
274 
275  //load module from file:
276  ModuleData modData(processHandle, hMods[counter], true);
277  ModuleScanReport *mappingScanReport = this->scanForMappingMismatch(modData, pReport);
278 
279  //load the original file to make the comparisons:
280  if (!modData.loadOriginal()) {
281  if (!args.quiet) {
282  std::cout << "[!][" << args.pid << "] Suspicious: could not read the module file!" << std::endl;
283  }
284  //make a report that finding original module was not possible
285  pReport.appendReport(new UnreachableModuleReport(processHandle, hMods[counter], 0, modData.szModName));
286  continue;
287  }
288 
289  // Don't scan modules that are in the ignore list
290  std::string plainName = peconv::get_file_name(modData.szModName);
291  if (is_in_list(plainName.c_str(), this->ignoredModules)) {
292  // ...but add such modules to the exports lookup:
293  if (pReport.exportsMap && modData.loadOriginal()) {
294  pReport.exportsMap->add_to_lookup(modData.szModName, (HMODULE)modData.original_module, (ULONGLONG)modData.moduleHandle);
295  }
296  if (!args.quiet) {
297  std::cout << "[*] Skipping ignored: " << std::hex << (ULONGLONG)modData.moduleHandle << " : " << modData.szModName << std::endl;
298  }
299  pReport.appendReport(new SkippedModuleReport(processHandle, modData.moduleHandle, modData.original_size, modData.szModName));
300  continue;
301  }
302  if (!args.quiet) {
303  std::cout << "[*] Scanning: " << modData.szModName << std::endl;
304  }
305  if (modData.isDotNet()) {
306  pReport.isManaged = true;
307 #ifdef _DEBUG
308  std::cout << "[*] Skipping a .NET module: " << modData.szModName << std::endl;
309 #endif
310  pReport.appendReport(new SkippedModuleReport(processHandle, modData.moduleHandle, modData.original_size, modData.szModName));
311  continue;
312  }
313  //load data about the remote module
314  RemoteModuleData remoteModData(processHandle, this->isReflection, hMods[counter]);
315  if (remoteModData.isInitialized() == false) {
316  //make a report that initializing remote module was not possible
317  pReport.appendReport(new MalformedHeaderReport(processHandle, hMods[counter], 0, modData.szModName));
318  continue;
319  }
320  t_scan_status is_hollowed = scanForHollows(processHandle, modData, remoteModData, pReport);
321  if (is_hollowed == SCAN_ERROR) {
322  continue;
323  }
324  if (is_hollowed == SCAN_NOT_SUSPICIOUS) {
325  //if the content does not differ, ignore the different name of the mapped file
326  mappingScanReport->status = SCAN_NOT_SUSPICIOUS;
327  }
328  if (pReport.exportsMap != nullptr) {
329  pReport.exportsMap->add_to_lookup(modData.szModName, (HMODULE) modData.original_module, (ULONGLONG) modData.moduleHandle);
330  }
331  // if hooks not disabled and process is not hollowed, check for hooks:
332  if (!args.no_hooks && (is_hollowed == SCAN_NOT_SUSPICIOUS)) {
333 
334  const bool scan_data = ((this->args.data >= pesieve::PE_DATA_SCAN_ALWAYS) && (this->args.data != pesieve::PE_DATA_SCAN_INACCESSIBLE_ONLY))
335  || (!this->isDEP && (this->args.data == pesieve::PE_DATA_SCAN_NO_DEP));
336 
337  const bool scan_inaccessible = (this->isReflection && (this->args.data >= PE_DATA_SCAN_INACCESSIBLE));
338 
339  scanForHooks(processHandle, modData, remoteModData, pReport, scan_data, scan_inaccessible);
340  }
341  }
342  return counter;
343 }
