58 HANDLE open_process(DWORD processID, bool reflection, bool quiet)
59 {
60 const DWORD basic_access = SYNCHRONIZE | PROCESS_VM_READ | PROCESS_QUERY_INFORMATION;
61 DWORD access = basic_access;
62 if (reflection) {
63 access |= pesieve::util::reflection_access | PROCESS_VM_OPERATION;
64 }
65
66 HANDLE hProcess = OpenProcess(access, FALSE, processID);
67
68 // if failed, try to open with the lower reflection access
69 if (!hProcess && access != basic_access && access != pesieve::util::reflection_access1) {
70 hProcess = OpenProcess(pesieve::util::reflection_access1, FALSE, processID);
71 }
72
73 // if failed, try to open with basic rights
74 if (!hProcess && access != basic_access) {
75 hProcess = OpenProcess( basic_access, FALSE, processID);
76 }
77
78 // check process compatibility
79 if (hProcess && !is_scanner_compatible(hProcess) && !quiet) {
80 util::print_in_color(WARNING_COLOR, "[!] Scanner mismatch! Try to use the 64bit version of the scanner!\n", true);
81 }
82
83 // opening succeeded, return the handle:
84 if (hProcess) {
85 return hProcess;
86 }
87
88 const DWORD last_err = GetLastError();
89
90 if (last_err == ERROR_ACCESS_DENIED) {
91 if (!quiet) {
92 std::cerr << "[-][" << processID << "] Could not open the process Error: " << last_err << std::endl;
93 //print more info:
94 check_access_denied(processID);
95 }
96
97 SetLastError(ERROR_ACCESS_DENIED);
98 throw std::runtime_error("Could not open the process: Access Denied");
99 return nullptr;
100 }
101 if (last_err == ERROR_INVALID_PARAMETER) {
102 if (!quiet) {
103 std::cerr << "-> Is this process still running?" << std::endl;
104 }
105 SetLastError(ERROR_INVALID_PARAMETER);
106 throw std::runtime_error("Could not open the process: Invalid Parameter");
107 }
108 return hProcess;

176pesieve::ReportEx* pesieve::scan_and_dump(IN const pesieve::t_params args)
177{
178 ReportEx *report = new(std::nothrow) ReportEx();
179 if (!report) {
180 // should not happen
181 return nullptr;
182 }
183 HANDLE orig_proc = nullptr; // original process handle
184 HANDLE cloned_proc = nullptr; // process reflection handle
185
186 if (!set_debug_privilege()) {
187 if (!args.quiet) std::cerr << "[-] Could not set debug privilege" << std::endl;
188 }
189
190 try {
191 orig_proc = open_process(args.pid, args.make_reflection, args.quiet);
192 HANDLE target_proc = orig_proc;
193
194 if (args.make_reflection) {
195 cloned_proc = make_process_reflection(orig_proc);
196 if (cloned_proc) {
197 target_proc = cloned_proc;
198 }
199 else {
200 if (!args.quiet) std::cerr << "[-] Failed to create the process reflection" << std::endl;
201 }
202 }
203
204 if (!args.quiet) {
205 if (cloned_proc) {
206 std::cout << "[*] Using process reflection!\n";
207 }
208 else {
209 std::cout << "[*] Using raw process!\n";
210 if (args.data == pesieve::PE_DATA_SCAN_INACCESSIBLE || args.data == pesieve::PE_DATA_SCAN_INACCESSIBLE_ONLY) {
211 print_in_color(WARNING_COLOR, "[WARNING] Scanning of inaccessible pages is possible only in reflection mode!\n");
212 }
213 }
214 }
215
216 const bool is_reflection = (cloned_proc) ? true : false;
217 ProcessScanner scanner(target_proc, is_reflection, args);
218 report->scan_report = scanner.scanRemote();
219
220 // dump process
221 if (report->scan_report) {
222 report->dump_report = make_dump(target_proc, is_reflection, args, *report->scan_report);
223 }
224 }
225 catch (std::exception &e) {
226 delete report;
227 report = nullptr;
228
229 if (!args.quiet) {
230 util::print_in_color(ERROR_COLOR, std::string("[ERROR] ") + e.what() + "\n", true);
231 }
232 }
233 if (cloned_proc) {
234 release_process_reflection(&cloned_proc);
235 }
236 CloseHandle(orig_proc);
237 return report;
238}
