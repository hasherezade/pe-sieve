38 static std::string FilterSymbolPath(const std::string& input, bool allowDownload)
39 {
40 std::string result;
41 size_t start = 0;
42
43 while (start < input.size())
44 {
45 size_t end = input.find(';', start);
46 if (end == std::string::npos) end = input.size();
47
48 std::string token = input.substr(start, end - start);
49
50 token.erase(0, token.find_first_not_of(" \t"));
51
52 size_t last = token.find_last_not_of(" \t");
53 if (last != std::string::npos)
54 token.erase(last + 1);
55 else
56 token.clear();
57
58 if (token.empty()) {
59 start = end + 1;
60 continue;
61 }
62 const bool isSrv = (_strnicmp(token.c_str(), "srv*", 4) == 0);
63
64 if (isSrv)
65 {
66 if (allowDownload)
67 {
68 // Keep full srv* entry
69 if (!result.empty()) result += ";";
70 result += token;
71 }
72 else
73 {
74 // Extract cache path: srv*<cache>*<server>
75 size_t first = token.find('*');
76 size_t second = token.find('*', first + 1);
77
78 if (first != std::string::npos && second != std::string::npos && second > first + 1)
79 {
80 std::string cache = token.substr(first + 1, second - first - 1);
81
82 if (!cache.empty()) {
83 if (!result.empty()) result += ";";
84 result += cache;
85 }
86 }
87 }
88 }
89 else
90 {
91 // Non-srv entries are always safe
92 if (!result.empty()) result += ";";
93 result += token;
94 }
95
96 start = end + 1;
97 }
98 return result;
99 }

101 static std::string BuildSymbolPath(bool enableAutoDownload)
102 {
103 const size_t bufferSize = 4096;
104 char envBuffer[bufferSize] = { 0 };
105
106 std::string path;
107
108 if (GetEnvironmentVariableA("_NT_SYMBOL_PATH", envBuffer, bufferSize)) {
109 path += FilterSymbolPath(envBuffer, enableAutoDownload);
110 }
111
112 if (GetEnvironmentVariableA("_NT_ALTERNATE_SYMBOL_PATH", envBuffer, bufferSize)) {
113 std::string filtered = FilterSymbolPath(envBuffer, enableAutoDownload);
114
115 if (!filtered.empty()) {
116 if (!path.empty()) path += ";";
117 path += filtered;
118 }
119 }
