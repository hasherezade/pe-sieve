59 const virtual void fieldsToJSON(std::stringstream &outs, size_t level, const pesieve::t_json_level &jdetails)
60 {
61 ModuleScanReport::_toJSON(outs, level);
62 outs << ",\n";
63 OUT_PADDED(outs, level, "\"thread_id\" : ");
64 outs << std::dec << tid;
65
66 outs << ",\n";
67 OUT_PADDED(outs, level, "\"indicators\" : [");
68 for (auto itr = indicators.begin(); itr != indicators.end(); ++itr) {
69 if (itr != indicators.begin()) {
70 outs << ", ";
71 }
72 outs << "\"" << indicator_to_str(*itr) << "\"";
73 }
74 outs << "]";
75 if (stack_ptr) {
76 outs << ",\n";
77 OUT_PADDED(outs, level, "\"stack_ptr\" : ");
78 outs << "\"" << std::hex << stack_ptr << "\"";
79 }
80 if (frames_count) {
81 outs << ",\n";
82 OUT_PADDED(outs, level, "\"frames_count\" : ");
83 outs << std::dec << frames_count;
84 }
85 if (thread_state != THREAD_STATE_UNKNOWN) {
86 outs << ",\n";
87 OUT_PADDED(outs, level, "\"thread_state\" : ");
88 outs << "\"" << translate_thread_state(thread_state) << "\"";
89
90 if (thread_state == THREAD_STATE_WAITING) {
91 outs << ",\n";
92 OUT_PADDED(outs, level, "\"thread_wait_reason\" : ");
93 outs << "\"" << translate_wait_reason(thread_wait_reason) << "\"";
94 }
95 }
96 if (susp_addr) {
97 outs << ",\n";
98 if (this->module && this->moduleSize) {
99 OUT_PADDED(outs, level, "\"susp_addr\" : ");
100 }
101 else {
102 OUT_PADDED(outs, level, "\"susp_return_addr\" : ");
103 }
104 outs << "\"" << std::hex << susp_addr << "\"";
105 }
106 if (this->module) {
107 outs << ",\n";
108 OUT_PADDED(outs, level, "\"protection\" : ");
109 outs << "\"" << std::hex << protection << "\"";
110 if (stats.isFilled()) {
111 outs << ",\n";
112 stats.toJSON(outs, level);
113 }
114 }
115 }
