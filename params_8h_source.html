207 bool fillStringParam(const std::string &paramId, PARAM_STRING &strparam)
208 {
209 StringParam* myStr = dynamic_cast<StringParam*>(this->getParam(paramId));
210 if (!myStr || !myStr->isSet()) {
211 return false;
212 }
213 std::string val = myStr->valToString();
214 const size_t len = val.length();
215 if (!len) {
216 return false;
217 }
218 alloc_strparam(strparam, len);
219 bool is_copied = false;
220 if (strparam.buffer) {
221 is_copied = copyCStr<StringParam>(paramId, strparam.buffer, strparam.length);
222 }
223 return is_copied;
224 }
