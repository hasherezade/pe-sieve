159WorkingSetScanReport* pesieve::WorkingSetScanner::scanRemote()
160{
161 MemPageData memPage(this->processHandle, this->pDetails.isReflection, this->memRegion.base, 0);
162 memPage.is_listed_module = this->processReport.hasModule(this->memRegion.base);
163
164 if (!memPage.isInfoFilled() && !memPage.fillInfo()) {
165#ifdef _DEBUG
166 std::cout << "[!] Could not fill: " << std::hex << memPage.start_va << " to: " << memPage.region_end << "\n";
167#endif
168 return nullptr;
169 }
170 // sanity checks to make sure that we are scanning the same page that was previously collected:
171 if (memPage.alloc_base != this->memRegion.alloc_base) {
172#ifdef _DEBUG
173 std::cerr << "WARNING: Alloc Base mismatch: " << std::hex << memPage.alloc_base << " vs " << this->memRegion.alloc_base << std::endl;
174#endif
175 return nullptr;
176 }
177 if ((memPage.region_end - memPage.region_start) != this->memRegion.size) {
178#ifdef _DEBUG
179 std::cerr << "WARNING: Size mismatch: " << std::hex << (memPage.region_end - memPage.region_start) << " vs " << this->memRegion.size << std::endl;
180#endif
181 return nullptr;
182 }
183
184 // is the page executable?
185 const bool is_any_exec = isExecutable(memPage);
186 if (!is_any_exec) {
187 // probably not interesting
188 return nullptr;
189 }
190
191 if (memPage.mapping_type == MEM_MAPPED && memPage.isRealMapping()) {
192 //probably legit
193 return nullptr;
194 }
195
196 if (memPage.mapping_type == MEM_IMAGE) {
197 memPage.loadModuleName();
198 memPage.loadMappedName();
199 if (!isScannedAsModule(memPage)) {
200 scanImg(memPage);
201 }
202 const size_t region_size = (memPage.region_end) ? (memPage.region_end - memPage.region_start) : 0;
203 if (this->processReport.hasModuleContaining(memPage.region_start, region_size)) {
204 // the area was already scanned
205 return nullptr;
206 }
207 }
208#ifdef _DEBUG
209 std::cout << std::hex << memPage.start_va << ": Scanning executable area" << std::endl;
210#endif
211 WorkingSetScanReport* my_report = this->scanExecutableArea(memPage);
212 if (!my_report) {
213 return nullptr;
214 }
215 my_report->is_executable = true;
216 my_report->protection = memPage.protection;
217 my_report->mapping_type = memPage.mapping_type;
218 my_report->mapped_name = memPage.mapped_name;
219 return my_report;
220}
