157WorkingSetScanReport* pesieve::WorkingSetScanner::scanRemote()
158{
159 if (!memPage.isInfoFilled() && !memPage.fillInfo()) {
160#ifdef _DEBUG
161 std::cout << "[!] Could not fill: " << std::hex << memPage.start_va << " to: " << memPage.region_end << "\n";
162#endif
163 return nullptr;
164 }
165 // is the page executable?
166 const bool is_any_exec = isExecutable(memPage);
167 if (!is_any_exec) {
168 // probably not interesting
169 return nullptr;
170 }
171
172 if (memPage.mapping_type == MEM_MAPPED && memPage.isRealMapping()) {
173 //probably legit
174 return nullptr;
175 }
176
177 if (memPage.mapping_type == MEM_IMAGE) {
178 memPage.loadModuleName();
179 memPage.loadMappedName();
180 if (!isScannedAsModule(memPage)) {
181 scanImg();
182 }
183 const size_t region_size = (memPage.region_end) ? (memPage.region_end - memPage.region_start) : 0;
184 if (this->processReport.hasModuleContaining(memPage.region_start, region_size)) {
185 // the area was already scanned
186 return nullptr;
187 }
188 }
189#ifdef _DEBUG
190 std::cout << std::hex << memPage.start_va << ": Scanning executable area" << std::endl;
191#endif
192 WorkingSetScanReport* my_report = this->scanExecutableArea(memPage);
193 if (!my_report) {
194 return nullptr;
195 }
196 my_report->is_executable = true;
197 my_report->protection = memPage.protection;
198 my_report->mapping_type = memPage.mapping_type;
199 my_report->mapped_name = memPage.mapped_name;
200 return my_report;
