131size_t pesieve::ProcessScanReport::countHdrsReplaced() const
132{
133 size_t replaced = 0;
134 const t_report_type type = REPORT_HEADERS_SCAN;
135
136 std::set<ModuleScanReport*>::iterator itr;
137 for (itr = this->reportsByType[type].begin(); itr != this->reportsByType[type].end(); ++itr) {
138 ModuleScanReport* report = *itr;
139 if (ModuleScanReport::get_scan_status(report) == SCAN_SUSPICIOUS) {

141 if (!hdrRep) continue; //it should not happen

144 replaced++;
145 }
146 }
147 }

151pesieve::t_report pesieve::ProcessScanReport::generateSummary() const
152{
153 t_report summary = { 0 };
154 summary.pid = this->pid;
155 summary.is_64bit = this->is64bit;
156 summary.is_managed = this->isManaged;
157 summary.is_reflection = this->isReflection;
158 summary.errors = static_cast<DWORD>(this->errorsCount);
159 summary.skipped = static_cast<DWORD>(this->reportsByType[REPORT_SKIPPED_SCAN].size());
160 summary.scanned = static_cast<DWORD>(this->reportsByType[REPORT_HEADERS_SCAN].size());
161
162 std::vector<ModuleScanReport*>::const_iterator itr = moduleReports.begin();
163 for (; itr != moduleReports.end(); ++itr) {
164 ModuleScanReport* report = *itr;
165 if (ModuleScanReport::get_scan_status(report) == SCAN_SUSPICIOUS) {
166 summary.suspicious++;
167 }
168 if (ModuleScanReport::get_scan_status(report) == SCAN_ERROR) {
169 summary.errors++;
170 }
171 }
172 summary.replaced = MASK_TO_DWORD(countHdrsReplaced());
173 summary.patched = MASK_TO_DWORD(countSuspiciousPerType(REPORT_CODE_SCAN));
174 summary.iat_hooked = MASK_TO_DWORD(countSuspiciousPerType(REPORT_IAT_SCAN));
175 summary.implanted_shc = MASK_TO_DWORD(countSuspiciousPerType(REPORT_MEMPAGE_SCAN) + countSuspiciousPerType(REPORT_THREADS_SCAN));
176 summary.implanted_pe = MASK_TO_DWORD(countSuspiciousPerType(REPORT_ARTEFACT_SCAN));
177 summary.implanted = MASK_TO_DWORD(summary.implanted_shc + summary.implanted_pe);
178 summary.hdr_mod = MASK_TO_DWORD(countSuspiciousPerType(REPORT_HEADERS_SCAN) - summary.replaced);
179 summary.unreachable_file = MASK_TO_DWORD(countSuspiciousPerType(REPORT_UNREACHABLE_SCAN) + countResultsPerType(REPORT_UNREACHABLE_SCAN, pesieve::SCAN_ERROR));
180 summary.other = MASK_TO_DWORD(summary.suspicious - (summary.patched + summary.replaced + summary.implanted + summary.hdr_mod + summary.iat_hooked));
181 return summary;
182}

184std::string pesieve::ProcessScanReport::listModules(size_t level, const pesieve::ProcessScanReport::t_report_filter &filter, const t_json_level &jdetails) const
185{
186 std::stringstream stream;
187 //summary:
188 OUT_PADDED(stream, level, "\"scans\" : [\n");
189 bool is_first = true;
190 std::vector<ModuleScanReport*>::const_iterator itr;
191 for (itr = this->moduleReports.begin(); itr != this->moduleReports.end(); ++itr) {
192 ModuleScanReport *mod = *itr;
193 if (is_shown_type(mod->status, filter)) {
194 if (!is_first) {
195 stream << ",\n";
196 }
197 OUT_PADDED(stream, level + 1, "{\n");
198 mod->toJSON(stream, level + 2, jdetails);
199 stream << "\n";
200 OUT_PADDED(stream, level + 1, "}");
201 is_first = false;
202 }
203 }
204 if (moduleReports.size()) {
205 stream << "\n";
206 }
207 OUT_PADDED(stream, level, "]\n");
208 return stream.str();
